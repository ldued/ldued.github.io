全站官网 head

调用样式uianhei.min.css

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<title>${cms.getTag("title")}</title>
${cms.getTag("keywords")}
<link href="/dist/css/uianhei.min.css" rel="stylesheet">
</head>

全站列表页&内容页 head

内容页需要做响应式适配，调用样式uianhei_m.min.css

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8">
<meta name="HandheldFriendly" content="True">
<meta name="MobileOptimized" content="320">
<meta content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=0" name="viewport">
<title>${cms.getTag("title")}</title>
${cms.getTag("keywords")}
<link href="/dist/css/uianhei_m.min.css" rel="stylesheet">
</head>

首页-推荐置顶

<#list cms.getSiteContents("content.channel.id in (3532,3533,607) and content.recommended= true",1) as a>

<p>${a.summary}</p>
</#list>

    <li class="top"><span>${a.publishDateStr?date?string("MM/dd")}</span><p>[${a.channel.cnName}]</p><a href="${a.publishUrl}" target="_blank" title="${a.title}"><#if a.title?length <23>${a.title}<#else>${a.title[0..21]}..</#if></a></li>
