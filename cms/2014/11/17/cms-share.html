main.backtop({
    scrollName: "scrollUp", /* 滚动id */

    scrollTarget: 0, /* 回滚距离顶部的位置 */
    zIndex: 1080 /* 按钮层级 */
});

<!--单张图轮播-->

    items : 1,
    animateOut: "fadeOut"
});

<!--单张图轮播-->
$("#carousel-2").owlCarousel({
    autoplay: false,
    autoplayTimeout: 8000,
    dots: false,
    nav: true,
    loop: true,
    margin: 10,
    items : 1
});

<!--多张图轮播 带左右按钮-->
var carousel3 = $("#carousel-3");
carousel3.on('initialized.owl.carousel', function() {
    $(this).find(".owl-prev").addClass("off");
})

carousel3.on('translated.owl.carousel', function() {
    var firstdiv = true,lastdiv = true,
        firstdiv = $(this).find(".owl-item").first().hasClass("active"),
        lastdiv  = $(this).find(".owl-item").last().hasClass("active");
        if(firstdiv){
            $(this).find(".owl-prev").addClass("off");
        }else{
            $(this).find(".owl-prev").removeClass("off");
        };
        if(lastdiv){
            $(this).find(".owl-next").addClass("off");
        }else{
            $(this).find(".owl-next").removeClass("off");
        };
})

carousel3.owlCarousel({
    autoplay: false,
    dots: false,
    nav: true,
    loop: false,
    margin: 20,
    items : 4,
    slideBy: 4
});

微信分享接口

WeixinApi.ready(function(Api) {
    // 微信分享的数据
    var wxData = {
        "appId": "", // 服务号可以填写appId
        "imgUrl" : "http://anhei.wanmei.com/images/share.jpg",//微博 多张图片使用||分隔
        "link" : "http://anhei.wanmei.com/main_m.htm",
        "desc" : "礼包领取、福利发放、攻略查询，就来《暗黑黎明》官方网站！",
        "title" : "《暗黑黎明》官方网站-十亿手机的暗黑梦"
    };

    // 用户点开右上角popup菜单后，点击分享给好友，会执行下面这个代码
    Api.shareToFriend(wxData);

    // 点击分享到朋友圈，会执行下面这个代码
    Api.shareToTimeline(wxData);

    // 点击分享到腾讯微博，会执行下面这个代码
    Api.shareToWeibo(wxData);
})
